\documentclass{article}
\usepackage[utf8]{inputenc}
\usepackage[margin=1in]{geometry}
\usepackage{graphicx}
\usepackage{hyperref}
\usepackage{booktabs}

\title{{ '{{' }} title or 'Business Report' {{ '}}' }}
\author{{ '{{' }} author or 'Report Author' {{ '}}' }}
\date{{ '{{' }} date or '\\today' {{ '}}' }}

\begin{document}

\maketitle

{% if abstract %}
\begin{abstract}
{{ abstract }}
\end{abstract}
{% endif %}

\section{Executive Summary}

{{ executiveSummary or 'This section provides an overview of the key findings and recommendations.' }}

{% if sections %}
{% for section in sections %}
\section{{ '{' }}{{ section.title }}{{ '}' }}

{{ section.content }}

{% if section.subsections %}
{% for subsection in section.subsections %}
\subsection{{ '{' }}{{ subsection.title }}{{ '}' }}
{{ subsection.content }}
{% endfor %}
{% endif %}

{% endfor %}
{% endif %}

{% if data %}
\section{Data Summary}

\begin{table}[ht]

\begin{tabular}{lc}

{% endfor %}

\end{tabular}

\end{table}
{% endif %}

\section{Conclusion}

{{ conclusion or 'This report provides valuable insights for decision making.' }}

{% for appendix in appendices %}

{% endif %}
